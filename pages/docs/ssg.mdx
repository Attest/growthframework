# Next.js SSG + MDX

import { useSSG } from 'nextra/ssg'

export const getStaticProps = ({ params }) => {
  return fetch('https://api.github.com/repos/shuding/nextra')
    .then(res => res.json())
    .then(repo => ({
      props: { ssg: {
        stars: repo.stargazers_count
      }},
      revalidate: 10
    }))
}

export const StarsRenderer = () => {
  const { stars } = useSSG()
  return <strong>{stars}</strong>
}

The [Nextra repository](https://github.com/shuding/nextra) has <StarsRenderer/> stars!

The page is prerendered at build time with `getStaticProps`. Check the source code for more information.
